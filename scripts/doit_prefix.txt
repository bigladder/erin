#!/bin/bash
date
if ! cmake -DERIN_TESTING=ON -DCMAKE_EXPORT_COMPILE_COMMANDS=ON -DCMAKE_BUILD_TYPE=${1:-Debug} ..; then
  echo CMAKE CALL FAILED!
  exit 1
fi
if ! cmake --build . -j 4; then
  echo BUILD FAILED!
  exit 1
fi
if ! ctest --output-on-failure; then
  echo CTEST FAILED!
  exit 1
fi
echo Running Regressions Suite...
if ! bundle exec ruby ../scripts/regression_test.rb; then
  echo REGRESSIONS FOUND!
  exit 1
fi
echo Running custom tests...
if ! ./bin/test_erin_devs_storage; then
  echo Failed test_erin_devs_storage!
  exit 1
fi
if ! ./bin/test_erin_devs_flow_limits; then
  echo Failed test_erin_devs_flow_limits!
fi
time ./bin/str8760
mkdir -p tidy
